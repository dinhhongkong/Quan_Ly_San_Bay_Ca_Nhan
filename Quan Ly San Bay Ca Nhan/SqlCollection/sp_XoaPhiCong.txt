USE [QL_SANBAY_CN]
GO

CREATE PROC sp_XoaPhiCong(
	@MANV int)
AS
BEGIN
	BEGIN TRANSACTION
		DECLARE @IDBangLai varchar(10)
		SELECT @IDBangLai = ID_BANGLAI FROM PHICONG WHERE IDNHANVIEN = @MANV

		DELETE FROM PHICONG WHERE IDNHANVIEN = @MANV
		DELETE FROM CT_BANGLAI WHERE ID_BANGLAI = @IDBangLai
	COMMIT
END
