CREATE PROC [dbo].[sp_ThemCaNhan]
	@OWNER_ID VARCHAR(10)
	, @TEN NVARCHAR(50)
	, @CMND NVARCHAR(13)
	, @DIACHI NVARCHAR(100)
	, @SDT VARCHAR(13)
AS
BEGIN
	BEGIN TRANSACTION
		INSERT INTO NGUOI(CMND, TEN, DIACHI, SDT) 
		VALUES(@CMND, @TEN, @DIACHI, @SDT)

		DECLARE @IDNGUOI INT
		SELECT @IDNGUOI = IDENT_CURRENT('NGUOI')

		INSERT INTO CHUMAYBAY VALUES(@OWNER_ID, 0)
		INSERT INTO CANHAN VALUES(@OWNER_ID, @IDNGUOI)
	COMMIT
END
