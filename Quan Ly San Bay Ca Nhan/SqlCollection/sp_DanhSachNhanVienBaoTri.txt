CREATE PROC sp_DanhSachNhanVienBaoTri
AS
BEGIN 
	SELECT NV.IDNHANVIEN, NV.IDNGUOI ,TEN, CMND, DIACHI, SDT, LUONG, CALAMVIEC, TRANGTHAINGHI
	FROM 
		NHANVIEN NV INNER JOIN NGUOI NG ON NV.IDNGUOI = NG.IDNGUOI
		WHERE 
			EXISTS (SELECT 'X' 
					FROM NHANVIEN NV_TEMP 
					INNER JOIN CT_NVBAOTRI NV_BT 
					ON NV_TEMP.IDNHANVIEN = NV_BT.IDNHANVIEN 
					WHERE NV.IDNHANVIEN = NV_TEMP.IDNHANVIEN)
END
