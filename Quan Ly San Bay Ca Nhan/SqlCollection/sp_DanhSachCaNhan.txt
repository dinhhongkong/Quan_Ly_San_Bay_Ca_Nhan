CREATE PROC [dbo].[sp_DanhSachCaNhan]
AS
SELECT OWNER_ID, N.IDNGUOI, CMND, TEN, DIACHI, SDT
FROM CANHAN C
JOIN NGUOI N
ON C.IDNGUOI = N.IDNGUOI
