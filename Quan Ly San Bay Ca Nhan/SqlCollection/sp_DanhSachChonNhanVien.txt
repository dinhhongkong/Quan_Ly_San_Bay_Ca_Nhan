CREATE PROC sp_DanhSachChonNhanVien
AS
BEGIN
SELECT NV.IDNHANVIEN, NV.IDNGUOI ,TEN, CMND, DIACHI, SDT, LUONG, CALAMVIEC, TRANGTHAINGHI
	FROM NHANVIEN NV INNER JOIN NGUOI NG ON NV.IDNGUOI = NG.IDNGUOI
	WHERE NV.IDNHANVIEN NOT IN (SELECT BT.IDNHANVIEN FROM CT_NVBAOTRI BT)
		AND NV.IDNHANVIEN NOT IN (SELECT PC.IDNHANVIEN FROM PHICONG PC)
END
